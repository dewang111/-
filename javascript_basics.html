JavaScript 基础语法

第一章：变量定义（let 与 const）

let：用于声明可重新赋值的变量
const：用于声明常量，声明后不可重新赋值

// ========== let 变量：可重新赋值 ========== let age = 25; // 声明一个变量并赋值 age = 30; // 可以修改值 console.log("年龄：", age); // 输出：年龄：30 // ========== const 常量：不可重新赋值 ========== const PI = 3.14159; // 声明一个常量 // PI = 3.14; // 错误！常量不能重新赋值 // ========== const 与对象的配合 ========== const person = { name: "小明", age: 20 }; person.age = 21; // 允许修改对象属性！ person.gender = "男"; // 允许添加新属性！ console.log(person); // { name: "小明", age: 21, gender: "男" }

第二章：变量类型（typeof 检测）

JavaScript 有 6 种基本类型：Number（数值）、String（字符串）、Boolean（布尔）、
null（空）、undefined（未定义），以及 Object（对象，包括 Array 数组）

// ========== 基本类型 ========== let num = 123; // Number 数值类型 let str = "Hello JavaScript"; // String 字符串类型 let isTrue = true; // Boolean 布尔类型（true/false） let nothing = null; // null 空值（注意：typeof null === "object" 是历史BUG） let notDef; // undefined 未定义（声明但未赋值） // ========== Object 对象类型 ========== let arr = [1, 2, 3]; // Array 数组（特殊对象） let obj = { x: 1, y: 2 }; // Object 普通对象 // ========== typeof 运算符：检测数据类型 ========== console.log(typeof num); // "number" console.log(typeof str); // "string" console.log(typeof isTrue); // "boolean" console.log(typeof nothing); // "object" （JS历史BUG，实际是null） console.log(typeof notDef); // "undefined" console.log(typeof arr); // "object" （数组本质是对象） console.log(typeof obj); // "object" console.log(typeof fn); // "function" （函数是特殊对象）

第三章：表达式（运算符）

表达式由值和运算符组成，用于计算结果。包括：算术运算符、比较运算符、逻辑运算符

// ========== 算术运算符 ========== let a = 10, b = 3; a + b; // 加法：13 a - b; // 减法：7 a * b; // 乘法：30 a / b; // 除法：3.333... a % b; // 取余（取模）：1（10除以3余1） a ** b; // 幂运算：1000（10的3次方） // ========== 比较运算符（返回布尔值） ========== a > b; // 大于：true a < b; // 小于：false a >= b; // 大于等于：true a <= b; // 小于等于：false a == b; // 宽松相等：false（只比较值，不比较类型） a != b; // 宽松不等：true a === b; // 严格相等：false（同时比较值和类型） a !== b; // 严格不等：true // ========== 逻辑运算符 ========== // && 与：两边都为true才返回true true && true; // true true && false; // false // || 或：任意一边为true就返回true true || false; // true false || false; // false // ! 非：取反 !true; // false !false; // true

第四章：流程控制（条件与循环）

流程控制用于决定代码的执行顺序：条件语句（if/else）和循环语句（for、while、do...while、for...of）

// ========== if / else 条件语句 ========== let score = 85; if (score >= 90) { console.log("优秀"); } else if (score >= 60) { console.log("及格"); } else { console.log("不及格"); } // ========== for 循环（已知循环次数） ========== // 语法：for (初始化; 条件; 增量) { 循环体 } for (let i = 0; i < 5; i++) { console.log("for循环第" + i + "次"); } // ========== while 循环（条件为真时重复） ========== // 语法：while (条件) { 循环体 } let count = 0; while (count < 3) { console.log("while循环：" + count); count++; } // ========== do...while 循环（至少执行一次） ========== // 语法：do { 循环体 } while (条件) let n = 0; do { console.log("do...while循环：" + n); n++; } while (n < 2); // ========== for...of 循环（遍历可迭代对象） ========== // 语法：for (变量 of 可迭代对象) { 循环体 } let fruits = ["苹果", "香蕉", "橙子"]; for (let fruit of fruits) { console.log("水果：" + fruit); }

第五章：函数（普通函数 与 箭头函数）

函数是一段可重复调用的代码块。普通函数使用 function 关键字声明，
箭头函数是 ES6 简洁语法，使用 => 符号。两者在 this 绑定上有区别。

// ========== 普通函数（function 关键字） ========== // 语法：function 函数名(参数) { 函数体 } function add(a, b) { return a + b; // return 返回结果 } add(3, 5); // 调用函数，返回 8 // ========== 箭头函数（ES6 新语法） ========== // 语法：(参数) => { 函数体 } // 当函数体只有一行时，可省略大括号和 return const multiply = (a, b) => a * b; multiply(4, 6); // 返回 24 // ========== 带条件判断的箭头函数 ========== const getGrade = (score) => { if (score >= 90) return "A"; if (score >= 80) return "B"; return "C"; }; // ========== 单参数的箭头函数可省略括号 ========== const double = x => x * 2; // 等同于 x => { return x * 2; }
